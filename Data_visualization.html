# Grouping by artist to get total streams
artist_streams = df.groupby('artist(s)_name')['streams'].sum().reset_index()

# Sorting and selecting the top 15 artists
top_artists = artist_streams.sort_values(by='streams', ascending=False).head(15)

# Bar Chart for Top 15 Artists by Stream Count
plt.figure(figsize=(20, 5))
plt.bar(top_artists['artist(s)_name'], top_artists['streams'], color='skyblue')
plt.title('Top 15 Artists by Stream Count')
plt.xlabel('Artist Name')
plt.ylabel('Total Streams')
plt.xticks(rotation=45)
plt.tight_layout()
plt.show()
